local UserInputService = game:GetService("UserInputService")
local TweenService = game:GetService("TweenService")
local RunService = game:GetService("RunService")
local Players = game:GetService("Players")
local Lighting = game:GetService("Lighting")
local LocalPlayer = Players.LocalPlayer
local Mouse = LocalPlayer:GetMouse()
local HttpService = game:GetService("HttpService")

local Blur = Instance.new("BlurEffect")
Blur.Name = "OrionBlur"
Blur.Size = 18
Blur.Parent = Lighting

local OrionLib = {
    Elements = {},
    ThemeObjects = {},
    Connections = {},
    Flags = {},
    Themes = {
        SciFiPurple = {
            Main = Color3.fromRGB(32,24,48),
            Second = Color3.fromRGB(48,36,72),
            Stroke = Color3.fromRGB(140,90,255),
            Divider = Color3.fromRGB(90,60,160),
            Text = Color3.fromRGB(235,225,255),
            TextDark = Color3.fromRGB(170,150,210)
        }
    },
    SelectedTheme = "SciFiPurple",
    Folder = nil,
    SaveCfg = false
}

local Icons = {}
pcall(function()
    Icons = HttpService:JSONDecode(game:HttpGetAsync("https://raw.githubusercontent.com/evoincorp/lucideblox/master/src/modules/util/icons.json")).icons
end)

local function GetIcon(i)
    return Icons[i]
end

local Orion = Instance.new("ScreenGui")
Orion.Name = "Orion"
Orion.Parent = gethui and gethui() or game.CoreGui

local function AddConnection(s,f)
    local c = s:Connect(f)
    table.insert(OrionLib.Connections,c)
    return c
end

local function Create(n,p,c)
    local o = Instance.new(n)
    for k,v in pairs(p or {}) do o[k]=v end
    for _,ch in pairs(c or {}) do ch.Parent=o end
    return o
end

local function ReturnProperty(o)
    if o:IsA("Frame") or o:IsA("TextButton") then return "BackgroundColor3" end
    if o:IsA("ScrollingFrame") then return "ScrollBarImageColor3" end
    if o:IsA("UIStroke") then return "Color" end
    if o:IsA("TextLabel") or o:IsA("TextBox") then return "TextColor3" end
    if o:IsA("ImageLabel") or o:IsA("ImageButton") then return "ImageColor3" end
end

local function AddThemeObject(o,t)
    OrionLib.ThemeObjects[t] = OrionLib.ThemeObjects[t] or {}
    table.insert(OrionLib.ThemeObjects[t],o)
    o[ReturnProperty(o)] = OrionLib.Themes[OrionLib.SelectedTheme][t]
    return o
end

local function SetTheme()
    for n,t in pairs(OrionLib.ThemeObjects) do
        for _,o in pairs(t) do
            o[ReturnProperty(o)] = OrionLib.Themes[OrionLib.SelectedTheme][n]
        end
    end
end

local MainWindow = AddThemeObject(Create("Frame",{
    Parent = Orion,
    Size = UDim2.new(0,615,0,344),
    Position = UDim2.new(0.5,-307,0.5,-172),
    BackgroundTransparency = 0.15,
    BorderSizePixel = 0,
    Active = true,
    Draggable = true
}),"Main")

AddThemeObject(Create("UICorner",{CornerRadius=UDim.new(0,12)},{MainWindow}),"Main")

local TopBar = AddThemeObject(Create("Frame",{
    Parent = MainWindow,
    Size = UDim2.new(1,0,0,50),
    BackgroundTransparency = 0.2,
    BorderSizePixel = 0
}),"Second")

AddThemeObject(Create("UIStroke",{
    Parent = MainWindow,
    Thickness = 1.4,
    Transparency = 0.2
}),"Stroke")

local Title = AddThemeObject(Create("TextLabel",{
    Parent = TopBar,
    Size = UDim2.new(1,-20,1,0),
    Position = UDim2.new(0,10,0,0),
    BackgroundTransparency = 1,
    Text = "Orion Sci‑Fi UI",
    Font = Enum.Font.GothamBlack,
    TextSize = 20,
    TextXAlignment = Left
}),"Text")

AddConnection(UserInputService.InputBegan,function(i)
    if i.KeyCode==Enum.KeyCode.RightShift then
        MainWindow.Visible = not MainWindow.Visible
        Blur.Enabled = MainWindow.Visible
    end
end)

AddConnection(Orion.Destroying,function()
    if Blur then Blur:Destroy() end
end)

SetTheme()

return OrionLib
